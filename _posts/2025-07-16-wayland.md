```shell
$ sudo apt update
$ sudo apt install weston

$ mkdir weston && cd weston
$ openssl genrsa -out tls.key 2048
$ openssl req -new -key tls.key -out tls.csr
$ openssl x509 -req -days 365 -signkey tls.key -in tls.csr -out tls.crt
```

Running (VNC backend):
```shell
$ weston \
--no-config \
--socket=/tmp/wl-test-env \
--backend=vnc \
--disable-transport-layer-security \
--vnc-tls-cert=$PWD/tls.crt \
--vnc-tls-key=$PWD/tls.key

$ env WAYLAND_DISPLAY=/tmp/wl-test-env wayland-info
$ env WAYLAND_DISPLAY=/tmp/wl-test-env weston-flower
```

Running (headless):
```shell
$ weston \
--no-config \
--socket=/tmp/wl-test-env \
--backend=headless \
--width=800 \
--height=600
```

Running application:
```shell
$ weston --shell=kiosk-shell.so -- $HOME/.local/bin/weston-simple-egl
```
